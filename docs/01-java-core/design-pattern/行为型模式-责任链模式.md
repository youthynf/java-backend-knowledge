# 行为型模式-责任链模式

行为型模式-责任链模式
将请求的发送者与接收者解耦，使多个接收者都有机会处理请求，请求沿责任链传递，直到被处理。请求需要多个对象处理（如审批流程、日志级别过滤）。

// 请求类
public class LeaveRequest {
    private int days; // 请假天数

    public LeaveRequest(int days) {
        this.days = days;
    }

    public int getDays() {
        return days;
    }
}

// 处理者抽象类
public abstract class Approver {
    protected Approver next; // 下一个处理者

    public Approver(Approver next) {
        this.next = next;
    }

    public abstract void approve(LeaveRequest request);
}

// 具体处理者：组长（处理≤3天）
public class TeamLeader extends Approver {
    public TeamLeader(Approver next) {
        super(next);
    }

    @Override
    public void approve(LeaveRequest request) {
        if (request.getDays() <= 3) {
            System.out.println("组长批准请假" + request.getDays() + "天");
        } else {
            next.approve(request); // 转发给下一级
        }
    }
}

// 具体处理者：经理（处理≤7天）
public class Manager extends Approver {
    public Manager(Approver next) {
        super(next);
    }

    @Override
    public void approve(LeaveRequest request) {
        if (request.getDays() <= 7) {
            System.out.println("经理批准请假" + request.getDays() + "天");
        } else {
            next.approve(request);
        }
    }
}

// 具体处理者：总监（处理≤30天）
public class Director extends Approver {
    public Director(Approver next) {
        super(next);
    }

    @Override
    public void approve(LeaveRequest request) {
        if (request.getDays() <= 30) {
            System.out.println("总监批准请假" + request.getDays() + "天");
        } else {
            System.out.println("请假天数过长，不批准");
        }
    }
}

// 使用示例
public class Client {
    public static void main(String[] args) {
        // 构建责任链：组长→经理→总监
        Approver chain = new TeamLeader(new Manager(new Director(null)));

        chain.approve(new LeaveRequest(2));  // 组长批准
        chain.approve(new LeaveRequest(5));  // 经理批准
        chain.approve(new LeaveRequest(20)); // 总监批准
        chain.approve(new LeaveRequest(40)); // 不批准
    }
}
