# 结构型模式-享元模式

结构型模式-享元模式
通过共享对象减少内存占用，适用于大量相似对象（如文字处理软件中的字符）。区分内部状态（可共享，如字符 'A'）和外部状态（不可共享，如位置、颜色）。

import java.util.HashMap;
import java.util.Map;

// 享元接口
public interface Character {
    void display(int x, int y); // 外部状态：位置
}

// 具体享元：字符
public class ConcreteCharacter implements Character {
    private char c; // 内部状态（可共享）

    public ConcreteCharacter(char c) {
        this.c = c;
    }

    @Override
    public void display(int x, int y) {
        System.out.println("显示字符 '" + c + "' 在位置 (" + x + "," + y + ")");
    }
}

// 享元工厂：管理共享对象
public class CharacterFactory {
    private Map<Character, Character> characters = new HashMap<>();

    // 获取享元对象（不存在则创建）
    public Character getCharacter(char c) {
        Character character = characters.get(c);
        if (character == null) {
            character = new ConcreteCharacter(c);
            characters.put(c, character);
        }
        return character;
    }
}

// 使用示例
public class Client {
    public static void main(String[] args) {
        CharacterFactory factory = new CharacterFactory();

        // 多次获取字符'A'，实际共享同一个对象
        Character a1 = factory.getCharacter('A');
        a1.display(10, 20);

        Character a2 = factory.getCharacter('A');
        a2.display(30, 40);

        System.out.println("是否为同一对象：" + (a1 == a2)); // true
    }
}
