# 行为型模式-迭代器模式

行为型模式-迭代器模式
提供一种方法顺序访问聚合对象（如集合）中的元素，而无需暴露其内部结构。

import java.util.NoSuchElementException;

// 迭代器接口
public interface Iterator<T> {
    boolean hasNext();
    T next();
}

// 聚合接口（可迭代对象）
public interface Aggregate<T> {
    Iterator<T> createIterator();
}

// 具体聚合：自定义列表
public class MyList<T> implements Aggregate<T> {
    private T[] elements;
    private int size;

    @SuppressWarnings("unchecked")
    public MyList(int capacity) {
        elements = (T[]) new Object[capacity];
    }

    public void add(T element) {
        elements[size++] = element;
    }

    public T get(int index) {
        return elements[index];
    }

    public int size() {
        return size;
    }

    @Override
    public Iterator<T> createIterator() {
        return new MyListIterator<>(this);
    }

    // 具体迭代器
    private static class MyListIterator<T> implements Iterator<T> {
        private MyList<T> list;
        private int index = 0;

        public MyListIterator(MyList<T> list) {
            this.list = list;
        }

        @Override
        public boolean hasNext() {
            return index < list.size();
        }

        @Override
        public T next() {
            if (!hasNext()) {
                throw new NoSuchElementException();
            }
            return list.get(index++);
        }
    }
}

// 使用示例
public class Client {
    public static void main(String[] args) {
        MyList<String> list = new MyList<>(3);
        list.add("A");
        list.add("B");
        list.add("C");

        Iterator<String> iterator = list.createIterator();
        while (iterator.hasNext()) {
            System.out.println(iterator.next()); // A → B → C
        }
    }
}
