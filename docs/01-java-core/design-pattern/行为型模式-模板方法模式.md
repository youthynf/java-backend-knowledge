# 行为型模式-模板方法模式

行为型模式-模板方法模式
定义一个操作中的算法骨架，将一些步骤延迟到子类中，使子类可以不改变算法结构而重定义特定步骤。关键要点：抽象类定义模板方法（固定流程）+ 抽象方法（子类实现可变步骤），模板方法用final修饰防止篡改流程。

// 抽象类（定义算法骨架）
public abstract class CaffeineBeverage {
    // 模板方法：固定流程，声明为final防止子类修改
    public final void prepareRecipe() {
        boilWater();       // 步骤1：煮水（固定）
        brew();            // 步骤2：冲泡（子类实现）
        pourInCup();       // 步骤3：倒入杯子（固定）
        addCondiments();   // 步骤4：加调料（子类实现）
    }

    // 抽象方法：由子类实现
    protected abstract void brew();
    protected abstract void addCondiments();

    // 具体方法：固定实现
    private void boilWater() {
        System.out.println("煮水");
    }

    private void pourInCup() {
        System.out.println("倒入杯子");
    }
}

// 具体子类：咖啡
public class Coffee extends CaffeineBeverage {
    @Override
    protected void brew() {
        System.out.println("用沸水冲泡咖啡");
    }

    @Override
    protected void addCondiments() {
        System.out.println("加糖和牛奶");
    }
}

// 具体子类：茶
public class Tea extends CaffeineBeverage {
    @Override
    protected void brew() {
        System.out.println("用沸水浸泡茶叶");
    }

    @Override
    protected void addCondiments() {
        System.out.println("加柠檬");
    }
}

// 使用示例
public class Client {
    public static void main(String[] args) {
        CaffeineBeverage coffee = new Coffee();
        System.out.println("准备咖啡：");
        coffee.prepareRecipe();
        // 输出：
        // 准备咖啡：
        // 煮水
        // 用沸水冲泡咖啡
        // 倒入杯子
        // 加糖和牛奶

        CaffeineBeverage tea = new Tea();
        System.out.println("\n准备茶：");
        tea.prepareRecipe();
        // 输出：
        // 准备茶：
        // 煮水
        // 用沸水浸泡茶叶
        // 倒入杯子
        // 加柠檬
    }
}
