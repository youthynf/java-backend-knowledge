# 创建型模式-抽象工厂模式

创建型模式-抽象工厂模式
提供一个接口，用于创建相关或依赖对象的家族（产品族），而无需指定具体类。关键要点：产品族接口 + 具体产品 + 抽象工厂 + 具体工厂，确保同一工厂创建的产品兼容（如 Windows 组件之间风格统一）。

// 产品接口族（按钮）
public interface Button {
    void render();
}

// 产品接口族（文本框）
public interface TextBox {
    void render();
}

// 具体产品：Windows按钮
public class WindowsButton implements Button {
    @Override
    public void render() {
        System.out.println("渲染Windows风格按钮");
    }
}

// 具体产品：Windows文本框
public class WindowsTextBox implements TextBox {
    @Override
    public void render() {
        System.out.println("渲染Windows风格文本框");
    }
}

// 具体产品：Mac按钮
public class MacButton implements Button {
    @Override
    public void render() {
        System.out.println("渲染Mac风格按钮");
    }
}

// 具体产品：Mac文本框
public class MacTextBox implements TextBox {
    @Override
    public void render() {
        System.out.println("渲染Mac风格文本框");
    }
}

// 抽象工厂接口（定义产品族的创建方法）
public interface UIFactory {
    Button createButton();
    TextBox createTextBox();
}

// 具体工厂：Windows工厂
public class WindowsUIFactory implements UIFactory {
    @Override
    public Button createButton() {
        return new WindowsButton();
    }

    @Override
    public TextBox createTextBox() {
        return new WindowsTextBox();
    }
}

// 具体工厂：Mac工厂
public class MacUIFactory implements UIFactory {
    @Override
    public Button createButton() {
        return new MacButton();
    }

    @Override
    public TextBox createTextBox() {
        return new MacTextBox();
    }
}

// 使用示例
public class Client {
    public static void main(String[] args) {
        UIFactory factory = new WindowsUIFactory(); // 可切换为MacUIFactory
        Button button = factory.createButton();
        TextBox textBox = factory.createTextBox();
        button.render();    // 输出：渲染Windows风格按钮
        textBox.render();   // 输出：渲染Windows风格文本框
    }
}
