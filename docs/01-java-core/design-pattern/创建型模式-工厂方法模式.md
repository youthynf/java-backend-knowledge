# 创建型模式-工厂方法模式

创建型模式-工厂方法模式
关键要点：产品接口 + 具体产品 + 工厂接口 + 具体工厂，新增产品只需扩展工厂和产品，符合开闭原则。

// 产品接口
public interface Document {
    void open();
}

// 具体产品：文本文档
public class TextDocument implements Document {
    @Override
    public void open() {
        System.out.println("打开文本文档");
    }
}

// 具体产品：表格文档
public class SpreadsheetDocument implements Document {
    @Override
    public void open() {
        System.out.println("打开表格文档");
    }
}

// 工厂接口
public interface DocumentFactory {
    Document createDocument();
}

// 具体工厂：文本文档工厂
public class TextDocumentFactory implements DocumentFactory {
    @Override
    public Document createDocument() {
        return new TextDocument();
    }
}

// 具体工厂：表格文档工厂
public class SpreadsheetDocumentFactory implements DocumentFactory {
    @Override
    public Document createDocument() {
        return new SpreadsheetDocument();
    }
}

// 使用示例
public class Client {
    public static void main(String[] args) {
        DocumentFactory factory = new TextDocumentFactory();
        Document doc = factory.createDocument();
        doc.open();  // 输出：打开文本文档
    }
}
