# 结构型模式-组合模式

结构型模式-组合模式
将对象组合成树形结构，使单个对象和组合对象的使用具有一致性（如文件系统的文件与文件夹）。

import java.util.ArrayList;
import java.util.List;

// 组件接口（统一节点和叶子的行为）
public interface FileSystemComponent {
    void printName();
    long getSize(); // 大小（字节）
}

// 叶子节点：文件
public class File implements FileSystemComponent {
    private String name;
    private long size;

    public File(String name, long size) {
        this.name = name;
        this.size = size;
    }

    @Override
    public void printName() {
        System.out.println(name);
    }

    @Override
    public long getSize() {
        return size;
    }
}

// 组合节点：文件夹（可包含子组件）
public class Folder implements FileSystemComponent {
    private String name;
    private List<FileSystemComponent> children = new ArrayList<>();

    public Folder(String name) {
        this.name = name;
    }

    public void addComponent(FileSystemComponent component) {
        children.add(component);
    }

    @Override
    public void printName() {
        System.out.println(name + "/");
        // 递归打印子组件
        for (FileSystemComponent child : children) {
            child.printName();
        }
    }

    @Override
    public long getSize() {
        // 递归计算总大小
        long total = 0;
        for (FileSystemComponent child : children) {
            total += child.getSize();
        }
        return total;
    }
}

// 使用示例
public class Client {
    public static void main(String[] args) {
        FileSystemComponent file1 = new File("a.txt", 1024);
        FileSystemComponent file2 = new File("b.jpg", 2048);

        Folder folder = new Folder("文档");
        folder.addComponent(file1);
        folder.addComponent(file2);

        folder.printName(); 
        // 输出：
        // 文档/
        // a.txt
        // b.jpg

        System.out.println("总大小：" + folder.getSize() + "字节"); // 3072字节
    }
}
