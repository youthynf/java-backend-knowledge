# 结构型模式-外观模式

结构型模式-外观模式
为子系统中的一组接口提供一个统一的高层接口，简化子系统的使用（如电脑启动只需按电源键，无需手动启动 CPU、内存、硬盘）。

// 子系统1：CPU
public class CPU {
    public void start() {
        System.out.println("CPU启动");
    }
}

// 子系统2：内存
public class Memory {
    public void start() {
        System.out.println("内存加载");
    }
}

// 子系统3：硬盘
public class HardDrive {
    public void start() {
        System.out.println("硬盘读取");
    }
}

// 外观类：提供简化接口
public class ComputerFacade {
    private CPU cpu;
    private Memory memory;
    private HardDrive hardDrive;

    public ComputerFacade() {
        this.cpu = new CPU();
        this.memory = new Memory();
        this.hardDrive = new HardDrive();
    }

    // 高层接口：一键启动电脑
    public void start() {
        cpu.start();
        memory.start();
        hardDrive.start();
        System.out.println("电脑启动完成");
    }
}

// 使用示例
public class Client {
    public static void main(String[] args) {
        ComputerFacade computer = new ComputerFacade();
        computer.start(); // 客户端无需关心子系统细节
        // 输出：
        // CPU启动
        // 内存加载
        // 硬盘读取
        // 电脑启动完成
    }
}
