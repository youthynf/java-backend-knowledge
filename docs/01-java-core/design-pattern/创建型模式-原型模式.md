# 创建型模式-原型模式

创建型模式-原型模式
通过复制（克隆）现有对象来创建新对象，而非通过new实例化，适用于创建成本高的场景。核心是实现Cloneable接口，重写clone()方法。

import java.util.ArrayList;
import java.util.List;

// 原型接口（实现Cloneable）
public class Document implements Cloneable {
    private String content;
    private List<String> images = new ArrayList<>();

    public Document(String content) {
        this.content = content;
        // 模拟初始化耗时操作（如读取文件）
        try { Thread.sleep(1000); } catch (InterruptedException e) {}
    }

    public void addImage(String image) {
        images.add(image);
    }

    // 重写clone()实现浅克隆（若需深克隆，需手动复制引用类型）
    @Override
    public Document clone() {
        try {
            Document clone = (Document) super.clone();
            // 深克隆：复制引用类型成员（避免共享同一List）
            clone.images = new ArrayList<>(this.images);
            return clone;
        } catch (CloneNotSupportedException e) {
            throw new AssertionError();
        }
    }

    // 测试
    public static void main(String[] args) {
        // 原始对象（创建耗时）
        Document original = new Document("初始内容");
        original.addImage("图1");

        // 克隆对象（无需重新初始化，效率高）
        Document clone = original.clone();
        clone.addImage("图2");

        System.out.println("原始文档图片：" + original.images); // [图1]
        System.out.println("克隆文档图片：" + clone.images);   // [图1, 图2]
    }
}
