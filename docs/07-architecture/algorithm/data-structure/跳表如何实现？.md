跳表如何实现？

package com.study.leet;

import java.util.Random;

/**
 * 跳表实现
 * 跳表是一种可以用来快速查找的数据结构，可以看作是对有序链表的优化
 * 
 * 1. 每个节点都有一个节点列表，每个节点列表的长度为当前跳表的最大层数，创建节点的时候随机生成。
 * 2. 每个节点的节点列表中，存储了当前节点在每一层的前进指针。
 * 3. 新增和删除核心是找到插入节点的每层的前一个节点，然后修改前一个节点的前进指针。
 * 4. 如果当前节点的层数比最高的节点还要大，则多出来的层需要更新头节点的指针。
 * 
 */
public class SkipList {
    // 节点内部类
    private static class Node {
        int value; // 节点存储的值
        Node[] forward; // 节点在各层的前进指针

        public Node(int value, int level) {
            this.value = value;
            // forward数组的大小为level+1，因为从0开始计数
            this.forward = new Node[level + 1];
        }
    }

    private static final int MAX_LEVEL = 32; // 最大层数
    private static final double P = 0.5; // 层级提升的概率
    private final Random random; // 随机数生成器
    private int level; // 当前跳表的最大层数
    private final Node head; // 头节点

    /**
     * 构造函数，初始化跳表
     */
    public SkipList() {
        this.random = new Random();
        this.level = 0;
        // 头节点不存储实际值
        this.head = new Node(-1, MAX_LEVEL);
    }

    /**
     * 生成随机层数
     * 使用概率P来决定是否增加层数
     */
    private int randomLevel() {
        int level = 0;
        // 以概率P增加层数，直到达到最大层数或概率不满足
        while (random.nextDouble() < P && level < MAX_LEVEL) {
            level++;
        }
        return level;
    }

    /**
     * 查找指定值是否存在于跳表中
     * @param value 要查找的值
     * @return 如果存在返回true，否则返回false
     */
    public boolean search(int value) {
        Node current = head;
        // 从当前最大层开始向下查找
        for (int i = level; i >= 0; i--) {
            // 在当前层向右查找，直到找到大于等于目标值的节点
            while (current.forward[i] != null && current.forward[i].value < value) {
                current = current.forward[i];
            }
        }
        // 移动到最底层的下一个节点，这应该是我们要找的节点
        current = current.forward[0];
        // 检查是否找到目标值
        return current != null && current.value == value;
    }

    /**
     * 向跳表中插入一个值
     * @param value 要插入的值
     */
    public void insert(int value) {
        // 创建update数组，用于记录每层需要更新的节点
        Node[] update = new Node[MAX_LEVEL + 1];
        Node current = head;

        // 从当前最大层开始向下查找，记录每层的前驱节点
        for (int i = level; i >= 0; i--) {
            while (current.forward[i] != null && current.forward[i].value < value) {
                current = current.forward[i];
            }
            update[i] = current;
        }

        // 生成新节点的随机层数
        int newLevel = randomLevel();
        // 如果新节点的层数大于当前最大层数，更新最大层数和update数组
        if (newLevel > level) {
            for (int i = level + 1; i <= newLevel; i++) {
                update[i] = head;
            }
            level = newLevel;
        }

        // 创建新节点
        Node newNode = new Node(value, newLevel);
        // 更新各层的指针，将新节点插入跳表
        for (int i = 0; i <= newLevel; i++) {
            newNode.forward[i] = update[i].forward[i];
            update[i].forward[i] = newNode;
        }
    }

    /**
     * 从跳表中删除一个值
     * @param value 要删除的值
     * @return 如果删除成功返回true，否则返回false
     */
    public boolean delete(int value) {
        // 创建update数组，用于记录每层需要更新的节点
        Node[] update = new Node[MAX_LEVEL + 1];
        Node current = head;

        // 从当前最大层开始向下查找，记录每层的前驱节点
        for (int i = level; i >= 0; i--) {
            while (current.forward[i] != null && current.forward[i].value < value) {
                current = current.forward[i];
            }
            update[i] = current;
        }

        // 移动到可能要删除的节点
        current = current.forward[0];

        // 检查是否找到了要删除的节点
        if (current != null && current.value == value) {
            // 更新各层的指针，绕过要删除的节点
            for (int i = 0; i <= level; i++) {
                // 如果前驱节点的下一个节点不是要删除的节点，说明当前层不需要更新
                if (update[i].forward[i] != current) {
                    break;
                }
                update[i].forward[i] = current.forward[i];
            }

            // 如果删除节点后，最高层没有节点了，降低最大层数
            while (level > 0 && head.forward[level] == null) {
                level--;
            }
            return true;
        }
        return false;
    }

    /**
     * 打印跳表的结构，用于调试
     */
    public void printList() {
        System.out.println("SkipList:");
        for (int i = 0; i <= level; i++) {
            Node node = head.forward[i];
            System.out.print("Level " + i + ": ");
            while (node != null) {
                System.out.print(node.value + " -> ");
                node = node.forward[i];
            }
            System.out.println("null");
        }
    }

    /**
     * 主方法，用于测试跳表的功能
     */
    public static void main(String[] args) {
        SkipList skipList = new SkipList();
        
        // 测试插入
        skipList.insert(3);
        skipList.insert(6);
        skipList.insert(7);
        skipList.insert(9);
        skipList.insert(12);
        skipList.insert(19);
        skipList.insert(17);
        skipList.insert(26);
        skipList.insert(21);
        skipList.insert(25);
        
        // 打印跳表结构
        skipList.printList();
        
        // 测试查找
        System.out.println("Search 19: " + skipList.search(19)); // 应该返回true
        System.out.println("Search 10: " + skipList.search(10)); // 应该返回false
        
        // 测试删除
        System.out.println("Delete 19: " + skipList.delete(19)); // 应该返回true
        System.out.println("Search 19 after delete: " + skipList.search(19)); // 应该返回false
        
        // 打印删除后的跳表结构
        skipList.printList();
    }
}
