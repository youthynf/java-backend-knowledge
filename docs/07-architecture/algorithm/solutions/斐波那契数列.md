# 斐波那契数列

斐波那契数列
避免使用递归，减少重复计算量：

package com.study.leet;

public class Fibonacci {

    /**
     * 计算第n个斐波那契数
     * @param n 第n个斐波那契数
     * @return 第n个斐波那契数的值
     */
    public int fibonacci(int n) {
        if (n <= 0) {
            return 0;
        }
        if (n == 1) {
            return 1;
        }
        int a = 0;
        int b = 1;
        for (int i = 2; i <= n; i++) {
            int temp = a + b;
            a = b;
            b = temp;
        }
        return b;
    }

}
